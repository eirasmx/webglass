Usage
// Basic usage
const glass = new LiquidGlass(element);

// With options
const glass = new LiquidGlass(element, {
  ior: 1.8,
  shape: 'squircle', // squircle | circle | lip | concave
  cornerRadius: 0.5,
  blur: 0.008,
  fresnel: 0.8,
});

// Update at runtime
glass.set({ ior: 2.0, blur: 0.01 });

// Remove
glass.destroy();
